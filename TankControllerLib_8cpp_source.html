118 void TankControllerLib::handleUI() {
119  COUT("TankControllerLib::handleUI() - " << state->name());
120  char key = Keypad_TC::instance()->getKey();
121  if (key == NO_KEY) {
122  if (!lastKeypadTime) {
123  // we have already reached an idle state, so don't do other checks
124  } else if (isInCalibration()) {
125  // we are in calibration, so don't return to main menu
126  } else if (nextState) {
127  // we already have a next state teed-up, do don't try to return to main menu
128  } else if (millis() - lastKeypadTime > IDLE_TIMEOUT) {
129  // time since last keypress exceeds the idle timeout, so return to main menu
130  setNextState((UIState *)new MainMenu(this));
131  lastKeypadTime = 0; // so we don't do this until another keypress!
132  }
133  } else {
134  serial("Keypad input: %c", key);
135  COUT("TankControllerLib::handleUI() - " << state->name() << "::handleKey(" << key << ")");
136  state->handleKey(key);
137  lastKeypadTime = millis();
138  }
139  updateState();
140  COUT("TankControllerLib::handleUI() - " << state->name() << "::loop()");
141  state->loop();
142 }
