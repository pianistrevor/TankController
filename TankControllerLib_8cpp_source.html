98 void TankControllerLib::handleUI() {
99  COUT("TankControllerLib::handleUI() - " << state->name());
100  char key = Keypad_TC::instance()->getKey();
101  if (key == NO_KEY) {
102  if (!lastKeypadTime) {
103  // we have already reached an idle state, so don't do other checks
104  } else if (isInCalibration()) {
105  // we are in calibration, so don't return to main menu
106  } else if (nextState) {
107  // we already have a next state teed-up, do don't try to return to main menu
108  } else if (millis() - lastKeypadTime > IDLE_TIMEOUT) {
109  // time since last keypress exceeds the idle timeout, so return to main menu
110  setNextState((UIState *)new MainMenu(this));
111  lastKeypadTime = 0; // so we don't do this until another keypress!
112  }
113  } else {
114  serial("Keypad input: %c", key);
115  COUT("TankControllerLib::handleUI() - " << state->name() << "::handleKey(" << key << ")");
116  state->handleKey(key);
117  lastKeypadTime = millis();
118  }
119  updateState();
120  COUT("TankControllerLib::handleUI() - " << state->name() << "::loop()");
121  state->loop();
122 }
