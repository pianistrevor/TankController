120void TankController::handleUI() {
121 COUT("TankController::handleUI() - " << state->name());
122 char key = Keypad_TC::instance()->getKey();
123 if (key == NO_KEY) {
124 if (!lastKeypadTime) {
125 // we have already reached an idle state, so don't do other checks
126 } else if (isInCalibration()) {
127 // we are in calibration, so don't return to main menu
128 } else if (nextState) {
129 // we already have a next state teed-up, do don't try to return to main menu
130 } else if (millis() - lastKeypadTime > IDLE_TIMEOUT) {
131 // time since last keypress exceeds the idle timeout, so return to main menu
132 setNextState((UIState *)new MainMenu(this));
133 lastKeypadTime = 0; // so we don't do this until another keypress!
134 }
135 } else {
136 serial(F("Keypad input: %c"), key);
137 COUT("TankController::handleUI() - " << state->name() << "::handleKey(" << key << ")");
138 state->handleKey(key);
139 lastKeypadTime = millis();
140 }
141 updateState();
142 COUT("TankController::handleUI() - " << state->name() << "::loop()");
143 state->loop();
144}

245void TankController::writeDataToSD() {
246 static uint32_t nextWriteTime = 0;
247 uint32_t msNow = millis();
248 COUT("nextWriteTime: " << nextWriteTime << "; now = " << msNow);
249 if (nextWriteTime > msNow) {
250 return;
251 }
252 char currentTemp[10];
253 char currentPh[10];
254 if (isInCalibration()) {
255 snprintf_P(currentTemp, sizeof(currentTemp), (PGM_P)F("C"));
256 snprintf_P(currentPh, sizeof(currentPh), (PGM_P)F("C"));
257 } else {
258 dtostrf((float)TempProbe_TC::instance()->getRunningAverage(), 4, 2, currentTemp);
259 dtostrf((float)PHProbe::instance()->getPh(), 5, 3, currentPh);
260 }
261 char targetTemp[10];
262 char targetPh[10];
263 dtostrf(TemperatureControl::instance()->getTargetTemperature(), 4, 2, targetTemp);
264 dtostrf(PHControl::instance()->getTargetPh(), 5, 3, targetPh);
265 static const char header[] = "time,tankid,temp,temp setpoint,pH,pH setpoint,onTime,Kp,Ki,Kd";
266 static const char format[] PROGMEM = "%02i/%02i/%4i %02i:%02i:%02i, %3i, %s, %s, %s, %s, %4lu";
267 char buffer[128];
268 DateTime_TC dtNow = DateTime_TC::now();
269 PID_TC *pPID = PID_TC::instance();
270 uint16_t tankId = EEPROM_TC::instance()->getTankID();
271 snprintf_P(buffer, sizeof(buffer), (PGM_P)format, (uint16_t)dtNow.month(), (uint16_t)dtNow.day(),
272 (uint16_t)dtNow.year(), (uint16_t)dtNow.hour(), (uint16_t)dtNow.minute(), (uint16_t)dtNow.second(),
273 (uint16_t)tankId, currentTemp, targetTemp, currentPh, targetPh, (unsigned long)(millis() / 1000));
274 strcpy_P(buffer + strnlen(buffer, sizeof(buffer)), (PGM_P)F(", "));
275 dtostrf(pPID->getKp(), 8, 1, buffer + strnlen(buffer, sizeof(buffer)));
276 strcpy_P(buffer + strnlen(buffer, sizeof(buffer)), (PGM_P)F(", "));
277 dtostrf(pPID->getKi(), 8, 1, buffer + strnlen(buffer, sizeof(buffer)));
278 strcpy_P(buffer + strnlen(buffer, sizeof(buffer)), (PGM_P)F(", "));
279 dtostrf(pPID->getKd(), 8, 1, buffer + strnlen(buffer, sizeof(buffer)));
280 SD_TC::instance()->appendData(header, buffer);
281 nextWriteTime = msNow / 1000 * 1000 + 1000; // round up to next second
282 COUT(buffer);
283}
