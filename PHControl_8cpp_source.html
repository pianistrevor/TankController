139void PHControl::updateControl(float pH) {
140 int msToBeOn;
141 int nowModWindow = millis() % WINDOW_SIZE;
142 uint32_t currentTime = DateTime_TC::now().secondstime();
143 switch (pHSetType) {
144 case FLAT_TYPE: {
145 currentPHTarget = targetPh;
146 break;
147 }
148 case RAMP_TYPE: {
149 if (currentTime < rampTimeEnd) {
150 currentPHTarget = rampStartingPh +
151 ((currentTime - rampTimeStart) * (targetPh - rampStartingPh) / (rampTimeEnd - rampTimeStart));
152 } else {
153 currentPHTarget = targetPh;
154 }
155 break;
156 }
157 case SINE_TYPE: {
158 uint32_t sineEndTime = sineStartTime + period;
159 if (currentTime >= sineEndTime) {
160 sineStartTime = DateTime_TC::now().secondstime();
161 sineEndTime = sineStartTime + period;
162 EEPROM_TC::instance()->setPhSineStartTime(sineStartTime);
163 }
164 float timeLeftTillPeriodEnd = sineEndTime - currentTime;
165 float percentNOTThroughPeriod = timeLeftTillPeriodEnd / period;
166 float percentThroughPeriod = 1 - percentNOTThroughPeriod;
167 float x = percentThroughPeriod * (2 * PI); // the x position for our sine wave
168 currentPHTarget = amplitude * sin(x) + targetPh; // y position in our sine wave
169 break;
170 }
171 default:
172 break;
173 }
174 COUT("PHControl::updateControl(" << pH << ") at " << millis());
175 if (usePID) {
176 msToBeOn = PID_TC::instance()->computeOutput(currentPHTarget, pH);
177 if (msToBeOn > 9000) {
178 if (msToBeOn > 10000 && lastWarnMS + 60000 < millis()) {
179 serial(F("WARNING: PID asked for an on time of %i which has been capped at 9000"), msToBeOn);
180 lastWarnMS = millis();
181 }
182 msToBeOn = 9000;
183 }
184 } else {
185 msToBeOn = pH > currentPHTarget ? 9000 : 0;
186 }
187 COUT("target: " << currentPHTarget << "; current: " << pH << "; nowModWindow = " << nowModWindow
188 << "; msToBeOn = " << msToBeOn);
189 bool newValue;
190 if (TankController::instance()->isInCalibration()) {
191 COUT("pH control should be off since in calibration");
192 newValue = TURN_SOLENOID_OFF; // turn off CO2 while in calibration
193 } else if (SOLENOID_OPENING_TIME < msToBeOn && nowModWindow < msToBeOn) {
194 COUT("pH control should be on");
195 newValue = TURN_SOLENOID_ON; // open CO2 solenoid
196 } else {
197 COUT("pH control should be off");
198 newValue = TURN_SOLENOID_OFF; // close CO2 solenoid
199 }
200 if (newValue != digitalRead(PH_CONTROL_PIN)) {
201 digitalWrite(PH_CONTROL_PIN, newValue);
202 uint32_t currentMS = millis();
203 serial(F("CO2 bubbler turned %s after %lu ms"), newValue ? "off" : "on", currentMS - lastSwitchMS);
204 lastSwitchMS = currentMS;
205 }
