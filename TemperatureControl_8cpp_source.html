141 COUT("Chiller::updateControl(" << currentTemperature << ") at " << currentMillis);

144 previousMillis = 0; // reset if clock went backwards (typical during tests)
145 }
146 // pause 30 seconds between switching chiller on and off to prevent damage to chiller
147 if (currentMillis - previousMillis < TIME_INTERVAL) {
148 COUT("Chiller update at " << currentMillis << " ignored due to update at " << previousMillis);
149 } else {

151 bool newValue;

158 // if the observed temperature is above the set-point range turn on the chiller

160 newValue = TURN_SOLENOID_ON;

163 // if the observed temperature is below the set-point range turn off the chiller
164 else if (currentTemperature <= currentTemperatureTarget - DELTA) {
