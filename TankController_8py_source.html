232 def updateDisplay(self):
233 # update Liquid Crystal display
234 self.lqd.SetLabelText(libTC.lcd(0) + '\n' + libTC.lcd(1))
235 # update EEPROM storage
236 for i, each in enumerate(self.eeprom):
237 each.SetLabelText('{:.4f}'.format(libTC.eeprom(i)))
238 # update Serial output
239 string = libTC.readSerial0().replace('\r\n', '\n')
240 if (string):
241 self.serial0.AppendText(string)
242 string = libTC.readSerial1().replace('\r\n', '\n')
243 if (string):
244 self.serial1.AppendText(string)
245 # update pins
246 self.pins.SetLabelText('LED: {}\nTmp: {}\nCO2: {}'.format(
247 'high' if libTC.readPin(13) else 'low',
248 'high' if libTC.readPin(47) else 'low',

275 def writeSD(self, e):
276 sd = os.path.join(os.getcwd(), 'SD')
277 if os.path.exists(sd):
278 shutil.rmtree(sd)
279 libTC.sdInit()
280 while sdPath := libTC.sdNextKey():
281 filePath = os.path.join(sd, sdPath[1:])
282 if not os.path.exists(os.path.dirname(filePath)):
283 try:
284 dirPath = os.path.dirname(filePath)
285 os.makedirs(dirPath)
286 except OSError as exc: # Guard against race condition
287 if exc.errno != errno.EEXIST:
288 raise
289 with open(filePath, "w") as f:
290 f.write(libTC.sdNextValue())
