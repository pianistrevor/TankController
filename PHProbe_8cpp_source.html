60void PHProbe::serialEvent1() {
61 // if we see that the Atlas Scientific product has sent a character
62 while (Serial1.available() > 0) {
63 String string = Serial1.readStringUntil('\r'); // read the string until we see a <CR>
64 if (string.length() > 0 && string[string.length() - 1] == '\r') {
65 // We should not see the CR (https://github.com/Arduino-CI/arduino_ci/pull/302)
66 string.remove(string.length() - 1);
67 }
68 if (string.length() > 0) {
69 if (isdigit(string[0])) { // if the first character in the string is a digit
70 // convert the string to a floating point number so it can be evaluated by the Arduino
71 value = string.toFloat();
72 // we have seen situations where the CO2 bubbler stays on and drives the pH down
73 if (value && value < 7.0) { // hang so as to trigger the watchdog timer reset
74 // treat 0 as valid since probe might not be connected
75 wdt_disable();
76 wdt_enable(WDTO_120MS); // allow enough time to print message
77 char buffer[50];
78 strncpy_P(buffer, (PGM_P)F("Triggering a reset because pH dropped to "), sizeof(buffer));
79 dtostrf(value, 5, 3, buffer + strnlen(buffer, sizeof(buffer)));
80 serial(buffer);
81 while (true) {
82 }
83 }
84 } else if (string[0] == '?') { // answer to a previous query
85 serial(F("PHProbe serialEvent1: \"%s\""), string.c_str());
86 if (string.length() > 7 && string.substring(0, 7) == "?SLOPE,") {
87 // for example "?SLOPE,16.1,100.0"
88 strncpy(slopeResponse, string.c_str(), sizeof(slopeResponse)); // Flawfinder: ignore
89 }
90 }
91 }
