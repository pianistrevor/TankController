218 void MainMenu::idle() {
219  PHControl *phControl = PHControl::instance();
220  char output[17];
221  output[0] = 'p';
222  output[1] = 'H';
223  output[2] = millis() / 1000 % 2 ? '=' : ' ';
224  dtostrf(PHProbe::instance()->getPh(), 5, 3, output + 3);
225  output[8] = ' ';
226  output[9] = phControl->isOn() ? 'B' : ' ';
227  output[10] = ' ';
228  dtostrf(PHControl::instance()->getTargetPh(), 5, 3, output + 11);
229  LiquidCrystal_TC::instance()->writeLine(output, 0);
230  TemperatureControl *tempControl = TemperatureControl::instance();
231  TempProbe_TC *tempProbe = TempProbe_TC::instance();
232  float temp = tempProbe->getRunningAverage();
233  if (temp < 0.0) {
234  temp = 0.0;
235  } else if (99.99 < temp) {
236  temp = 99.99;
237  }
238  char status = tempControl->isHeater() ? 'h' : 'c';
239  if (tempControl->isOn()) {
240  status = toupper(status);
241  }
242  output[0] = 'T';
243  output[1] = millis() / 1000 % 2 ? '=' : ' ';
244  dtostrf(temp, 5, 2, output + 2);
245  output[7] = ' ';
246  output[8] = status;
247  output[9] = ' ';
248  dtostrf(tempControl->getTargetTemperature(), 5, 2, output + 10);
249  LiquidCrystal_TC::instance()->writeLine(output, 1);
250 }
