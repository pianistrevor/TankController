54bool EthernetServer_TC::file() {
55 // Buffer has something like "GET /path HTTP/1.1"
56 // and we want to put a null at the end of the path.
57 int i = 4;
58 while (buffer[i] != ' ') {
59 ++i;
60 }
61 buffer[i] = '\0';
62 // Look for file in SD card.
63 if (!SD_TC::instance()->exists(buffer + 4)) {
64 serial(F("file - \"%s\" not found!"), buffer + 4);
65 return false;
66 }
67 // Open file and send headers with file size.
68 File file = SD_TC::instance()->open(buffer + 4);
69 uint32_t size = file.size();
70 serial(F("file \"%s\" has a size of %lu"), buffer + 4, size);
71 sendHeadersWithSize(size);
72 // Send file contents
73 uint32_t totalBytes = 0;
74 uint32_t timeInRead = 0;
75 uint32_t timeInWrite = 0;
76 uint32_t timeInFlush = 0;
77
78 wdt_disable();
79 uint32_t flushCount = 0;
80 while (file.available32()) {
81 uint32_t startTime = millis();
82 int readSize = file.read(buffer, sizeof(buffer)); // Flawfinder: ignore
83 timeInRead += millis() - startTime;
84 startTime = millis();
85 int writeSize = client.write(buffer, readSize);
86 timeInWrite += millis() - startTime;
87 if (writeSize != readSize) {
88 serial(F("totalBytes = %lu; read = %d; write = %d"), totalBytes, readSize, writeSize);
89 break;
90 }
91 totalBytes += writeSize;
92 if (totalBytes % 8196 == 0) {
93 startTime = millis();
94 client.flush();
95 timeInFlush += millis() - startTime;
96 ++flushCount;
97 }
98 }
99 file.close();
100 client.stop();
101 state = NOT_CONNECTED;
102 serial(F("write = %lu; freeMemory = %i"), totalBytes, TankController::instance()->freeMemory());
103 serial(F("timeInRead = %lu; timeInWrite = %lu; timeInFlush = %lu"), timeInRead, timeInWrite, timeInFlush);
104 wdt_enable(WDTO_8S);
105 return true;
106}

183int EthernetServer_TC::weekday(int year, int month, int day) {
184 // Calculate day of week in proleptic Gregorian calendar. Sunday == 0.
185 int adjustment, mm, yy;
186 if (year < 2000)
187 year += 2000;
188 adjustment = (14 - month) / 12;
189 mm = month + 12 * adjustment - 2;
190 yy = year - adjustment;
191 return (day + (13 * mm - 1) / 5 + yy + yy / 4 - yy / 100 + yy / 400) % 7;
192}
